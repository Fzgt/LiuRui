## 6 May 字母异位词分组 Lc49
```ts
// 思路：哈希表 字符串排序
function groupAnagrams(strs: string[]): string[][] {
    const map = new Map()
    strs.forEach(item => {
        const temp = item.split('').sort().join()
        if(map.has(temp)) {
            const current = map.get(temp)
            current.push(item)
            map.set(temp, current)
        } else {
            map.set(temp, [ item ])
        }
    })
    return Array.from(map.values())
};
```

## 9 May 最长连续序列 Lc128
```ts
// 思路：类似滑动窗口
// 执行时间击败66% 内存占用击败96%
function longestConsecutive(nums: number[]): number {
    if(!nums.length) {
        return 0
    }
    nums.sort((a, b) => a - b)
    let res = 1
    let right = 0
    let temp = 1
    while(right < nums.length) {
        if(nums[right+1] === nums[right] + 1) {
            temp += 1
        } else if (nums[right+1] === nums[right]) {
          while(nums[right+1] !== nums[right]) {
            temp += 1
            right ++
            continue
          }
        } else {
            res = Math.max(temp, res)
            temp = 1
        }
        right ++
    }
    return res
};
```

## 15 May 移动零 Lc283
```ts
// 思路：双指针
// 执行时间击败11% 内存占用击败91%
function moveZeroes(nums: number[]): void {
    let left = 0
    let right = 0
    while (right < nums.length) {
        if (nums[left] === 0) {
            right = left + 1
            while (right < nums.length && nums[right] === 0) {
                right ++
            }
            if(right < nums.length) {
                [nums[left], nums[right]] = [nums[right], nums[left]]
            }
        } else {
            right ++
        }
        left ++
    }
};
```

## 20 May 盛水最多的容器 Lc11
```ts
// 思路：双指针 两边收缩
// 执行时间击败5.28% 内存占用击败99.98%
function maxArea (height: number[]): number {
    let square = 0
    let res = 0
    let left = 0
    let right = height.length - 1 
    while(left < right) {
        square = (right - left) * Math.min(height[left], height[right])
        res = Math.max(res, square)
        height[left] < height[right] ? left++ : right--
    }
    return res
};
```

## 25 May 三数之和 Lc15
```ts
// 思路：双指针 两边收缩 左指针随外层循环改变
// 执行时间击败58.03% 内存占用击败45.09%
function threeSum(nums: number[]): number[][] {
    nums.sort((a, b) => a - b)
    const res = []
    for(let i=0;i<nums.length;i++) {
        const current = nums[i]
        if(current > 0 || (i-1 >= 0 && current === nums[i-1])) {
            continue
        }
        let left = i + 1
        let right = nums.length - 1
        while(left < right) {
            if(current + nums[left] + nums[right] === 0) {
                res.push([current, nums[left], nums[right]])
                while(nums[left] === nums[left+1]){
                    left ++
                }
                while(nums[right] === nums[right-1]) {
                    right --
                }
                left ++
                right --
            } else if(current + nums[left] + nums[right] < 0) {
                left ++
            } else {
                right --
            }
        }
    }
    return res
};
```
